Then crop:

```bash
node -e "
const sharp = require('sharp');
sharp('logo.png')
  .metadata()
  .then(m => {
    // The monogram occupies roughly the left ~45% of the image (top portion).
    // We crop a centered square that captures only the tcw mark.
    const w = m.width, h = m.height;
    const cropSize = Math.min(w, h) * 0.5;
    const left = Math.round(w * 0.20);
    const top = Math.round(h * 0.10);
    return sharp('logo.png').extract({ left, top, width: Math.round(cropSize), height: Math.round(cropSize) }).resize(512, 512).png().toFile('public/logo/logo-mark.png');
  })
  .then(() => console.log('mark done'));
"
```

Then derive favicon (32×32) and apple-touch-icon (180×180):

```bash
node -e "
const sharp = require('sharp');
Promise.all([
  sharp('public/logo/logo-mark.png').resize(32, 32).png().toFile('public/logo/favicon.png'),
  sharp('public/logo/logo-mark.png').resize(180, 180).png().toFile('public/logo/apple-touch-icon.png'),
]).then(() => console.log('icons done'));
"
```

If the crop coordinates produce a poorly-framed monogram, adjust `left`/`top` values manually. The cropped image must show ONLY the `tcw` monogram, centered, with reasonable padding.
